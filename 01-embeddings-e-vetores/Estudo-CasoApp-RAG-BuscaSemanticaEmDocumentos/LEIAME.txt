# Estudo de Caso - App de RAG Para Busca Semântica em Documentos - Aplicação e Uso de Modelos de Embeddings e Bancos de Dados Vetoriais

# Abra o terminal ou prompt de comando, navegue até a pasta com os arquivos e execute o comando abaixo para criar um ambiente virtual:

conda create --name dsamebvec python=3.13

# Ative o ambiente:

conda activate dsamebvec (ou: source activate dsamebvec)

# Instale o pip e as dependências:

conda install pip
pip install -r requirements.txt 

# Execute o script:

streamlit run dsa_app.py

# Acesse o jupyter pelo seu navegador padrão.

# Use os comandos abaixo para desativar o ambiente virtual e remover o ambiente (opcional):

conda deactivate
conda remove --name dsamebvec --all
